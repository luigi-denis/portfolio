APACHE NGINX

CONFIGURATION APACHE

ports.conf
Listen 5501
Listen 80
<IfModule ssl_module>
        Listen 5502
        Listen 443
</IfModule>

<IfModule mod_gnutls.c>
        Listen 5502
        Listen 443
</IfModule>
vhost admin apache.txt
<VirtualHost 150.10.0.10:5501>
        ServerName admin.rainbowbank.com
        Redirect permanent / https://150.10.0.10:5502
</VirtualHost>

<VirtualHost 150.10.0.10:5502>
        ServerName admin.rainbowbank.com
        ServerAlias 150.10.0.10
        DocumentRoot /var/www/admin.rainbowbank.com
        SSLEngine On
        SSLCertificateFile /srv/ssl/cert/rainbowbank.crt
        SSLCertificateKeyFile /srv/ssl/key/rainbowbank.key
</VirtualHost>
vhost extranet apache.txt
<VirtualHost 192.168.10.50:80>
    ServerName extranet.rainbowbank.com
    Redirect permanent / https://192.168.10.50:443
</VirtualHost>

<VirtualHost 192.168.10.50:443>
    ServerName extranet.rainbowbank.com
    DocumentRoot /var/www/extranet.rainbowbank.com
    SSLEngine On
    SSLCertificateFile /srv/ssl/cert/rainbowbank.crt
    SSLCertificateKeyFile /srv/ssl/key/rainbowbank.key
</VirtualHost>
module apache ssl.txt
<IfModule mod_ssl.c>
        SSLRandomSeed startup builtin
        SSLRandomSeed startup file:/dev/urandom 512
        SSLRandomSeed connect builtin
        SSLRandomSeed connect file:/dev/urandom 512

        SSLCipherSuite HIGH:!aNULL
        SSLProtocol all -SSLv3
</IfModule>
module apache evasive.txt
<IfModule mod_evasive20.c>
    DOSHashTableSize    3097
    DOSPageCount        2
    DOSSiteCount        50
    DOSPageInterval     1
    DOSSiteInterval     1
    DOSBlockingPeriod   10
    DOSSystemCommand \"/bin/echo %s >> /var/log/apache2/mod_evasive/dos_evasive.log && /bin/date >> /var/log/apache2/mod_evasive/dos_evasive.log\"
</IfModule>
module reqtimeout apache.txt
<IfModule reqtimeout_module>
        RequestReadTimeout header=5-10,minrate=500
        RequestReadTimeout body=10,minrate=500
</IfModule>

CONFIGURATION NGINX

vhost admin nginx.txt
server {
    listen 150.10.0.10:5501;
    server_name 150.10.0.10;
    return 301 https://$server_name:5502$request_uri;
}

server {
    listen 150.10.0.10:5502 ssl;
    server_name 150.10.0.10;
    root /var/www/admin.rainbowbank.com;
    ssl_certificate /srv/ssl/rainbowbank.crt;
    ssl_certificate_key /srv/ssl/rainbowbank.key;
}
vhost extranet nginx.txt
server {
    listen 192.168.10.51:80;
    server_name extranet.rainbowbank.com;
    return 301 https://192.168.10.51$request_uri;
}

server {
    listen 192.168.10.51:443 ssl;
    server_name extranet.rainbowbank.com;
    root /var/www/extranet.rainbowbank.com;
    ssl_certificate /srv/ssl/rainbowbank.crt;
    ssl_certificate_key /srv/ssl/rainbowbank.key;
}

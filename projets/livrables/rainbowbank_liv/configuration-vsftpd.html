# Certificats SSL

rsa_private_key_file=/srv/ssl/key/rainbowbank.key

# Accès utilisateurs locaux
local_enable=YES
write_enable=YES
local_umask=022
userlist_enable=YES
userlist_file=/etc/vsftpd.userlist
userlist_deny=NO

# Répertoires racine

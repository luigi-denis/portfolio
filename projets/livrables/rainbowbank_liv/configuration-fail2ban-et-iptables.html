# Autoriser trafic Web sur réseau Extranet

# Rejeter le reste du trafic 150.10.0.0/16 avec ICMP unreachable

-A INPUT -s 192.168.10.0/24 -j DROP

# Autoriser les sorties vers réseau d'administration
-A OUTPUT -d 150.10.0.0/16 -o ens34 -p tcp -m multiport --sports 20,21,22,989,990,5501,5502 -j ACCEPT

# Autoriser les sorties vers réseau Extranet
-A OUTPUT -d 192.168.10.0/24 -p tcp -m multiport --dports 80,443 -j ACCEPT

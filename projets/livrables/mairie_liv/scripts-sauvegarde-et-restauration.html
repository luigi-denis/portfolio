inbackup.sh
Stratégie 1 : Sauvegarde incrémentale avec rotation LIVE/OLD.
#!/bin/bash

# Répertoire source
SRC_DIR="/home/viewer/FICHIERS/"

# Répertoire de destination
DEST_DIR="view@192.168.1.109:/home/view/FICHIERS/LIVE"

# Fichier de log
LOG_FILE="/home/viewer/inbackup.log"

# Début de log 
echo "Début de sauvegarde du répertoire FIHCHIERS à $(date)" >> $LOG_FILE

# Purge des données du répertoire OLD sur le serveur distant
ssh view@192.168.1.109 "rm -rf /home/view/FICHIERS/OLD/*" >> $LOG_FILE 2>&1

# Copie du contenu de LIVE vers OLD sur le serveur distant
ssh view@192.168.1.109 "cp -r /home/view/FICHIERS/LIVE/* /home/view/FICHIERS/OLD/" >> $LOG_FILE 2>&1

# Exécution de la sauvegarde incrémentale avec rsync
rsync -avzr --delete -e ssh $SRC_DIR $DEST_DIR >> $LOG_FILE 2>&1

# Fin de log
echo "Sauvegarde du répertoire FIHCHIERS terminée à $(date)" >> $LOG_FILE

restauration.sh
Script interactif de restauration de fichiers ou de machines virtuelles.
#!/bin/bash

# Initialisation des variables
repertoire_base="/home/view"
repertoire_sauvegarde=""
sous_repertoire=""
repertoire_complet="$repertoire_base/$repertoire_sauvegarde/$sous_repertoire"
source_restauration=""
repertoire_destination="/viewer/$repertoire_sauvegarde/$source_restauration"
serveur_sauvegarde="view@192.168.1.109"

# Fonction pour demander le choix de backup
demander_choix_de_backup() {
    echo "Choisissez une option :"
    echo "1. Restaurer une VM"
    echo "2. Restaurer des fichiers"
    echo "3. Quitter le script"
    read -p "Votre choix : " choix

    case $choix in
        1)
            repertoire_sauvegarde="MACHINES"
            sous_repertoire=""
            backup
            ;;
        2)
            repertoire_de_backup
            ;;
        3)
            echo "Quitter le script."
            exit 0
            ;;
        *)
            echo "Choix invalide. Veuillez réessayer."
            demander_choix_de_backup
            ;;
    esac
}

# Fonction pour demander le choix de répertoire de backup
repertoire_de_backup() {
    echo "Choisissez une option de répertoire :"
    echo "1. TICKETS"
    echo "2. SITE"
    echo "3. FICHIERS"
    echo "4. RH"
    echo "5. MAILS"
    echo "6. Revenir en arrière"
    read -p "Votre choix : " choix

    case $choix in
        1) repertoire_sauvegarde="TICKETS" ;;
        2) repertoire_sauvegarde="SITE" ;;
        3) repertoire_sauvegarde="FICHIERS" ;;
        4) repertoire_sauvegarde="RH" ;;
        5) repertoire_sauvegarde="MAILS" ;;
        6) demander_choix_de_backup ;;
        *)
            echo "Choix invalide. Veuillez réessayer."
            repertoire_de_backup
            ;;
    esac

    sous_repertoire
}

# Fonction pour demander le choix du sous-répertoire
sous_repertoire() {
    echo "Choisissez une option de sous-répertoire :"
    echo "1. Avant dernière sauvegarde (OLD)"
    echo "2. Dernière sauvegarde (LIVE)"
    echo "3. Revenir en arrière"
    read -p "Votre choix : " choix

    case $choix in
        1) sous_repertoire="OLD" ;;
        2) sous_repertoire="LIVE" ;;
        3) repertoire_de_backup ;;
        *)
            echo "Choix invalide. Veuillez réessayer."
            sous_repertoire
            ;;
    esac

    backup
}

# Fonction pour effectuer le backup
backup() {
    repertoire_complet="$repertoire_base/$repertoire_sauvegarde/$sous_repertoire"
    echo "Vous avez choisi le répertoire suivant : $repertoire_complet"
    
    # Afficher l'arborescence du répertoire via SSH
    ssh "$serveur_sauvegarde" "tree -FDs $repertoire_complet"
    
    read -p "Indiquez le chemin du fichier à restaurer : " source_restauration
    
    read -p "Confirmez-vous la restauration ? (1. Oui / 2. Non / 3. Revenir en arrière) : " confirmation

    case $confirmation in
        1)
            repertoire_destination="/viewer/$repertoire_sauvegarde/$source_restauration"
            rsync -avz --delete -e ssh "$serveur_sauvegarde:$repertoire_complet/$source_restauration" "/home/viewer/$repertoire_sauvegarde/$source_restauration" >> /home/viewer/restauration.log 2>&1
            echo "Restauration effectuée. Détails enregistrés dans /home/viewer/restauration.log"
            echo "$(date) : $repertoire_complet/$source_restauration -> /home/viewer/$repertoire_sauvegarde/$source_restauration" >> /home/viewer/restauration.log
            echo "Restauration terminée à $(date)" >> /home/viewer/restauration.log
            ;;
        2)
            backup
            ;;
        3)  
            demander_choix_de_backup
            ;;
        *)
            echo "Choix invalide. Veuillez réessayer."
            backup
            ;;
    esac
}

# Lancement du script
demander_choix_de_backup
